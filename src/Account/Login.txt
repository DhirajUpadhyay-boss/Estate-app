Login
import React, { useState } from 'react';
import { Phone } from 'lucide-react';
import { Link, useNavigate } from 'react-router-dom';
import { useUser } from '../context/UserContext.jsx';
import { auth } from '../context/firebase';

const Login = () => {
  const [phone, setPhone] = useState('');
  const [error, setError] = useState('');
  const { user, markLoggedIn } = useUser();

  const navigate = useNavigate();

  const handleLogin = (e) => {
    e.preventDefault();
    setError('');

    const cleanedInput = phone.replace(/\D/g, '');
    const normalizedInput = cleanedInput.slice(-10);

    if (normalizedInput.length !== 10) {
      setError('Please enter a valid 10-digit phone number.');
      return;
    }

    if (!user) {
      setError('No account found on this device. Please register first.');
      console.log('Navbar state:', { user, isLoggedIn });
      setTimeout(() => navigate('/register'), 800);
      return;
    }

    if (normalizedInput === user.phone) {
      alert('Login successful!');
      
      markLoggedIn();          // ⬅ this is what makes the profile logo appear
      navigate('/projects');   // or '/profile' if you prefer
    } else {
      setError('Phone number not recognized. Please register with this number first.');
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-blue-100 flex items-center justify-center p-4">
      <div className="max-w-md w-full">
        <div className="text-center mb-8">
          <h1 className="text-4xl font-bold text-gray-800 mb-2">Welcome back!</h1>
          <p className="text-gray-600">Login with your registered phone number</p>
        </div>

        <div className="bg-white rounded-2xl shadow-xl p-8">
          <form onSubmit={handleLogin} className="space-y-6">
            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">
                Phone Number
              </label>
              <div className="relative">
                <Phone className="absolute left-4 top-4 text-gray-400" size={20} />
                <input
                  type="tel"
                  placeholder="10-digit number"
                  value={phone}
                  onChange={(e) =>
                    setPhone(e.target.value.replace(/\D/g, '').slice(0, 10))
                  }
                  maxLength={10}
                  className="w-full pl-12 pr-4 py-4 border-2 rounded-xl focus:border-blue-500 focus:ring-4 focus:ring-blue-100 outline-none"
                />
              </div>
            </div>

            {error && (
              <div className="p-3 rounded-lg bg-red-50 border border-red-200 text-sm text-red-700">
                {error}
              </div>
            )}

            <button
              type="submit"
              className="w-full bg-blue-600 text-white py-4 rounded-xl font-bold hover:bg-blue-700 transition"
            >
              Login
            </button>

            <p className="text-center text-sm text-gray-600">
              New here?{' '}
              <Link to="/register" className="text-blue-600 font-semibold hover:underline">
                Register
              </Link>
            </p>
          </form>
        </div>

        <div className="mt-8 text-center text-sm text-gray-600">
          Secure access with your registered phone number
        </div>
      </div>
    </div>
  );
};

export default Login;
